**New file: `scripts/spectrum-approval.sh`**
```bash
#!/usr/bin/env bash
# Spectrum PreToolUse approval gate.
# Called by the PreToolUse hook before each spectrum worker tool invocation.
# Writes an approval request, waits up to 30s for controller response, auto-approves.
#
# Environment (provided by Claude Code hook runner):
#   HOOK_TOOL_NAME  — name of tool being called
#   HOOK_SESSION_ID — spectrum worker session ID
#   HOOK_STORY_ID   — extracted from session tag (spectrum-worker-<story-id>)
#
set -euo pipefail

APPROVAL_DIR="${PRISM_PROJECT_DIR:-.}/.prism/local/spectrum-approvals"
STORY_ID="${HOOK_STORY_ID:-unknown}"
TOOL="${HOOK_TOOL_NAME:-unknown}"
REQUEST_ID="$(date +%s%N)"

mkdir -p "$APPROVAL_DIR/$STORY_ID"

REQUEST_FILE="$APPROVAL_DIR/$STORY_ID/$REQUEST_ID.request"
APPROVE_FILE="$APPROVAL_DIR/$STORY_ID/$REQUEST_ID.approve"
DENY_FILE="$APPROVAL_DIR/$STORY_ID/$REQUEST_ID.deny"

# Write approval request
echo "{\"tool\": \"$TOOL\", \"story\": \"$STORY_ID\", \"ts\": \"$(date -Iseconds)\"}" > "$REQUEST_FILE"

# Wait up to 30s for controller response
TIMEOUT=30
elapsed=0
while [[ $elapsed -lt $TIMEOUT ]]; do
    if [[ -f "$APPROVE_FILE" ]]; then
        rm -f "$REQUEST_FILE" "$APPROVE_FILE"
        exit 0  # Approved — tool call proceeds
    fi
    if [[ -f "$DENY_FILE" ]]; then
        rm -f "$REQUEST_FILE" "$DENY_FILE"
        exit 1  # Denied — tool call blocked
    fi
    sleep 1
    elapsed=$((elapsed + 1))
done

# Timeout — auto-approve
rm -f "$REQUEST_FILE"
exit 0
```

  ```python
  def discover_stale_versions(root: Path, new_version: str) -> list[tuple[Path, int, str]]:
      """
      Grep the repo for version-string patterns that don't match new_version.
      Returns list of (path, line_number, matched_version).
      Excludes: .git/, node_modules/, .prism/shared/ref/, .prism/shared/evals/v[0-9]*
      File types: .md, .json, .go, .ts, .tsx
      """
      import subprocess
      EXCLUDE_DIRS = [".git", "node_modules", ".prism/shared/ref"]
      # Dynamic exclusion for eval version dirs
      evals_dir = root / ".prism" / "shared" / "evals"
      if evals_dir.exists():
          for d in evals_dir.iterdir():
              if d.is_dir() and re.match(r"v\d+", d.name):
                  EXCLUDE_DIRS.append(str(d.relative_to(root)))
      
      semver_pattern = r'\b\d+\.\d+\.\d+\b'
      stale = []
      extensions = {".md", ".json", ".go", ".ts", ".tsx"}
      
      for path in root.rglob("*"):
          if not path.is_file():
              continue
          if path.suffix not in extensions:
              continue
          if any(excl in str(path) for excl in EXCLUDE_DIRS):
              continue
          try:
              text = path.read_text(encoding="utf-8", errors="ignore")
          except Exception:
              continue
          for lineno, line in enumerate(text.splitlines(), 1):
              for match in re.finditer(semver_pattern, line):
                  found = match.group()
                  if found != new_version:
                      stale.append((path.relative_to(root), lineno, found))
      return stale
  ```
